üìÑ Document Node
The root of the entire DOM tree. Represents the whole HTML document.
Example 1: Finding Elements
// Get element by ID
let header = document.getElementById('main-header');
// Get element by class name
let buttons = document.getElementsByClassName('btn');
// Get first element matching CSS selector
let firstParagraph = document.querySelector('p');

Example 2: Creating Elements
// Create new elements
let newDiv = document.createElement('div');
let newParagraph = document.createElement('p');
let newImage = document.createElement('img');
// Set properties and append to document
newDiv.textContent = 'Hello World';
document.body.appendChild(newDiv);

Example 3: Document Properties
// Access document properties
console.log(document.title); // Page title
console.log(document.URL); // Current URL
console.log(document.doctype); // Document type
console.log(document.documentElement); // HTML element
